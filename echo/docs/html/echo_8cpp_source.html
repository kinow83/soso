79 void Echo::consoleLog(int lv, const char *fmt, va_list ap, uint8_t *hex,
80  size_t len, const char *color, bool bold) {
81  struct tm t_now;
82  time_t now;
83  char strtime[256];
84  char buf[LOG_BUFSIZE];
85  const char *lv_str;
86  int n, i;
87  FILE *fp = NULL;
88  static char init_env = 0;
89 
90  if (_log_level >= lv) {
91  if (_filename) {
92  fp = fopen(_filename, "a+");
93  assert(fp != NULL);
94  } else {
95  fp = stdout;
96  }
97 
98  if (init_env == 0) {
99  // os default file buffering.
100  setvbuf(fp, (char *)NULL, _IOLBF, (size_t)0);
101  init_env = 1;
102  }
103 
104  lv_str = LOG_STR[lv];
105  now = time(NULL);
106  localtime_r(&now, &t_now);
107 
108  strftime(strtime, sizeof(strtime), "%H:%M:%S", &t_now);
109  n = vsnprintf(buf, sizeof(buf), fmt, ap);
110 
111  if (hex) {
112  for (i = 0; i < (int)len; i++) {
113  n += snprintf(buf + n, sizeof(buf) - n, " %02x", hex[i]);
114  }
115  }
116 
117  if (bold == true) {
118  fprintf(fp, "%s", VTC_BOLD);
119  }
120  if (hex) {
121  fprintf(fp, "%s%s: %s: (len:%ld) %s\n", color, strtime, lv_str, len, buf);
122  } else {
123  fprintf(fp, "%s%s: %s: %s\n", color, strtime, lv_str, buf);
124  }
125  fprintf(fp, "%s", VTC_RESET);
126 
127  if (_exit_on_fatal && (lv == L_FATAL)) {
128  exit(1);
129  }
130 
131  if (fp != stdout) {
132  fclose(fp);
133  }
134  }
135 }
